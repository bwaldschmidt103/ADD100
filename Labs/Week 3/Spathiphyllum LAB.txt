user_input = input("Please enter the plant you want: ").strip()

Spathiphyllum = user_input.lower() == 'spathiphyllum'
spathiphyllum = user_input.lower() == 'small spathiphyllum'

if Spathiphyllum:
    print("Yes - Spathiphyllum is the best plant ever!")
elif spathiphyllum:
    print("No, I want a big Spathiphyllum!")
else:
    print(f"Spathiphyllum! Not {user_input}!")
